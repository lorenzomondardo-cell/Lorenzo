// Sequência de regras normais antes do limite do trote const baseRules = [ { id: 1, description: "Sua senha deve ter pelo menos 5 caracteres.", check: (t) => t.length >= 5 }, { id: 2, description: "Sua senha deve incluir um número.", check: (t) => /\d/.test(t) }, { id: 3, description: "Sua senha deve incluir uma letra maiúscula.", check: (t) => /[A-Z]/.test(t) }, { id: 4, description: "Sua senha deve incluir um caractere especial (ex: !, @, #).", check: (t) => /[^A-Za-z0-9]/.test(t) }, { id: 5, description: "Sua senha precisa ter pelo menos 3 números.", check: (t) => (t.match(/\d/g) || []).length >= 3 }, { id: 6, description: "Sua senha precisa ter pelo menos 2 letras maiúsculas.", check: (t) => (t.match(/[A-Z]/g) || []).length >= 2 } ]; function updateRules() { const password = passwordInput.value; rulesContainer.innerHTML = ''; // GATILHO DO TROTE: Se a senha passar de 25 caracteres, o jogo acaba com a piada if (password.length >= 25) { passwordInput.disabled = true; // Bloqueia o campo para a pessoa não digitar mais const trollDiv = document.createElement('div'); trollDiv.classList.add('troll-message'); trollDiv.innerHTML = "❌ ERRO CRÍTICO:
Sua senha ficou muito grande! Diminua o tamanho para garantir a segurança. (Atual: " + password.length + " caracteres).

😜 Pegadinha! Você caiu no trote!"; rulesContainer.appendChild(trollDiv); return; } let allPreviousPassed = true; for (let rule of baseRules) { if (!allPreviousPassed) break; const isPassed = rule.check(password); const ruleDiv = document.createElement('div'); ruleDiv.classList.add('rule'); ruleDiv.classList.add(isPassed ? 'valid' : 'invalid'); ruleDiv.innerText = `Regra ${rule.id}: ${rule.description}`; rulesContainer.appendChild(ruleDiv); if (!isPassed) { allPreviousPassed = false; } } // Se ele passou de todas as regras básicas mas ainda não atingiu o limite de tamanho do trote, // criamos uma regra infinita para forçar a senha a crescer até estourar o limite de 25 caracteres. if (allPreviousPassed && password.length > 0) { const currentLength = password.length; const targetLength = currentLength + 4; // Pede sempre mais caracteres do que ele tem atualmente const ruleDiv = document.createElement('div'); ruleDiv.classList.add('rule', 'invalid'); ruleDiv.innerText = `Regra Avançada: Para maior segurança, sua senha agora precisa ter pelo menos ${targetLength} caracteres (Atual: ${currentLength}).`; rulesContainer.appendChild(ruleDiv); } } passwordInput.addEventListener('input', updateRules); updateRules();
